求定积分近似值

初步建立

#include <iostream>
#include <cmath>
using namespace std;    
double fx(double x=0)    //定义被积函数
{
    double s;
     s=pow(2,x);    //在此修改函数
    return s;

double gx(double a=0,double b=0,int n=1)    //定义简单积分，先默认a<=b
{
    int i=1;double l=(b-a)/n;double s=0;
    for(i;i<=n;i++)
        s+=fx(a+i*l);
    return s;
}    
int main()    //加入防错系统
{
    double a=0;double b=0;int n=1;
    cout<<"请输入积分下限，上限，分割份数:";
    cin>>a>>b>>n;
    if(a<=b)
        cout<<"所求积分为:"<<gx(a,b,n)<<endl;
    else
        cout<<"所求积分为:"<<-gx(b,a,n)<<endl;
    return 0;
}

由于使用过剩近似值代替，误差较大，应做出改进

改进近似值

#include <iostream>
#include <cmath>
using namespace std;    
double fx(double x=0)    //定义被积函数
{
    double s;
     s=pow(2,x);    //在此修改函数
    return s;
}    
double gx(double a=0,double b=0,int n=1)    //定义简单积分，先默认a<=b,过剩
{
    int i=1;double l=(b-a)/n;double s=0;
    for(i;i<=n;i++)
        s+=fx(a+i*l);
    s*=l;
    return s;
}    
double hx(double a=0,double b=0,int n=1)    //定义简单积分，先默认a<=b，不足
{
    int i=0;double l=(b-a)/n;double s=0;
    for(i;i<n;i++)
        s+=fx(a+i*l);
    s*=l;
    return s;
}    
int main()    //加入防错系统
{
    double a=0;double b=0;int n=1;
    cout<<"请输入积分下限，上限，分割份数:";
    cin>>a>>b>>n;
    if(a<=b)
        cout<<"所求积分在  "<<hx(a,b,n)<<"到  "<<gx(a,b,n)<<"之间,平均值为  "<<(hx(a,b,n)+gx(a,b,n))/2<<endl;
    else
        cout<<"所求积分在  "<<-hx(b,a,n)<<"到  "<<-gx(b,a,n)<<"之间,平均值为  "<<-(hx(b,a,n)+gx(b,a,n))/2<<endl;
    return 0;
}

改进：比较近似值大小

#include <iostream>
#include <cmath>
using namespace std;    
double fx(double x=0)    //定义被积函数
{
    double s;
     s=pow(2,x);    //在此修改函数
    return s;
}    
double gx(double a=0,double b=0,int n=1)    //定义简单积分，先默认a<=b,过剩
{
    int i=1;double l=(b-a)/n;double s=0;
    for(i;i<=n;i++)
        s+=fx(a+i*l);
    s*=l;
    return s;
}    
double hx(double a=0,double b=0,int n=1)    //定义简单积分，先默认a<=b，不足
{
    int i=0;double l=(b-a)/n;double s=0;
    for(i;i<n;i++)
        s+=fx(a+i*l);
    s*=l;
    return s;
}    
int main()    //加入防错系统
{
    double a=0;double b=0;int n=1;
    cout<<"请输入积分下限，上限，分割份数:";
    cin>>a>>b>>n;
    if(a<=b)
    {
        if(hx(a,b,n)<gx(a,b,n))
            cout<<"所求积分在  "<<hx(a,b,n)<<"到  "<<gx(a,b,n)<<"之间,平均值为  "<<(hx(a,b,n)+gx(a,b,n))/2<<endl;
        else
            cout<<"所求积分在  "<<gx(a,b,n)<<"到  "<<hx(a,b,n)<<"之间,平均值为  "<<(hx(a,b,n)+gx(a,b,n))/2<<endl;
    }
    else
    {
        if(hx(b,a,n)>gx(b,a,n))
            cout<<"所求积分在  "<<-hx(b,a,n)<<"到  "<<-gx(b,a,n)<<"之间,平均值为  "<<-(hx(b,a,n)+gx(b,a,n))/2<<endl;
        else
            cout<<"所求积分在  "<<-gx(b,a,n)<<"到  "<<-hx(b,a,n)<<"之间,平均值为  "<<-(hx(b,a,n)+gx(b,a,n))/2<<endl;
    }
    return 0;
}
